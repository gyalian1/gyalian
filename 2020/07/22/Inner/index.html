Inner

内部类语法

1、InnerTool称为内部类
Outer称为外部类
先实例化再调用内部类，实例化只能在Outer内完成
package com.youkeda.util;
public class OuterUtil {
public class InnerTool { // 内部类
public int add(int a,int b){
return a+b;
}
}
private InnerTool tool = new InnerTool();
}

2、内部实例化
实现3个数字的加法
package com.youkeda.util;
public class OuterUtil {
public class InnerTool { // 内部类
public int add(int a,int b){
return a+b;
}
}
private InnerTool tool = new InnerTool();
public int add(int a,int b,int c){
return tool.add( tool.add(a,b) , c );
}
}

test测试类：
package com.youkeda.test;
import com.youkeda.util.OuterUtil;
public class OuterUtilTest {
public static void main(String[] args){
OuterUtil outer = new OuterUtil();
int sum = outer.add(1,2,3);
System.out.println(sum);
}
}

3、访问实例内部类
OuterUtil.InnerTool tool = new OuterUtil().new InnerTool();

4、静态内部类
内部类也可以声明静态内部类，只要增加一个static关键词
package com.youkeda.util;
public class OuterUtil {
public static class InnerTool { // 内部类
public int add(int a,int b){
return a+b;
}
}
}

如果声明为静态内部类后，就可以直接调用内部类，不需要实例外部类
package com.youkeda.test;
import com.youkeda.util.OuterUtil;
// 静态内部类是需要单独 import 的
import com.youkeda.util.OuterUtil.InnerTool;
public class OuterUtilTest {
public static void main(String[] args){
InnerTool tool = new InnerTool();
int sum = tool.add(1,2);
System.out.println(sum);
}
}

5、局部内部类
在方法内部定义一个类，这个类就是局部内部类
public class Tester {
public void test(){
class B{
int a;
}
B b = new B();
}
}

6、例子
package com.youkeda.test;
import com.youkeda.util.OuterUtil;

public class InnerTest {
public static void main (String[] args) {

OuterUtil outer = new OuterUtil();

int sum = outer.add(12,13,25);

int sub = outer.less(80,40);

System.out.println("加法的结果是:"+sum);
System.out.println("减法的结果是:"+sub);

}

}

package com.youkeda.util;
public class OuterUtil {
public class InnerTool{
public int add(int a,int b){
return a+b;
}
public int less(int a,int b){
return a-b;
}
}
private InnerTool tool = new InnerTool();
public int add(int a,int b,int c){
return tool.add(tool.add(a,b) , c);
}

public int less(int a,int b){
return tool.less(a,b);
}
}
